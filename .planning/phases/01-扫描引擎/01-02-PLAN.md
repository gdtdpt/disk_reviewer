From RESEARCH.md code examples (Pattern: 逻辑盘枚举):

```rust
use windows::Win32::System::SystemInformation::GetLogicalDrives;
use windows::Win32::Storage::FileSystem::GetDiskFreeSpaceExW;
use windows::PCWSTR;

#[derive(Debug, Clone)]
pub struct DriveInfo {
    pub letter: char,
    pub total_bytes: u64,
    pub free_bytes: u64,
    pub used_bytes: u64,
}

pub fn enumerate_drives() -> Vec<DriveInfo> {
    let bitmask = unsafe { GetLogicalDrives() };
    let mut drives = Vec::new();
    for i in 0..26 {
        if bitmask & (1 << i) != 0 {
            let letter = (b'A' + i as u8) as char;
            let path = format!(r"{}:\", letter);
            let mut total: u64 = 0;
            let mut free: u64 = 0;
            let wide: Vec<u16> = path.encode_utf16().chain(std::iter::once(0)).collect();
            let ok = unsafe {
                GetDiskFreeSpaceExW(
                    PCWSTR(wide.as_ptr()),
                    None,
                    Some(&mut total),
                    Some(&mut free),
                )
            };
            if ok.is_ok() {
                drives.push(DriveInfo {
                    letter,
                    total_bytes: total,
                    free_bytes: free,
                    used_bytes: total - free,
                });
            }
        }
    }
    drives
}
```

**关键点**:
- 使用 `GetLogicalDrives()` 位掩码遍历 A-Z（0-25 位）
- 路径格式: `X:\`（不需要 `\\?\` 前缀，因为盘根路径很短）

- `GetDiskFreeSpaceExW` 的返回值用 `is_ok()` 判断，失败则跳过该盘
- `used_bytes = total - free`（不是从 API 直接获取）

```rust
use crate::platform::drives::{self, DriveInfo};

pub struct DiskReviewerApp {
    pub drives: Vec<DriveInfo>,
}

impl DiskReviewerApp {
    pub fn new(cc: &eframe::CreationContext<'_>) -> Self {
        let drives = drives::enumerate_drives();
        Self { drives }
    }
}

impl eframe::App for DiskReviewerApp {
    fn update(&mut self, ctx: &egui::Context, _frame: &mut eframe::Frame) {
        egui::CentralPanel::default().show(ctx, |ui| {
            ui.heading("Disk Reviewer");
            ui.separator();
            ui.label("逻辑盘列表:");
            for drive in &self.drives {
                ui.label(format!(
                    "{}:  总计 {:.1} GB  已用 {:.1} GB  可用 {:.1} GB",
                    drive.letter,
                    drive.total_bytes as f64 / 1e9,
                    drive.used_bytes as f64 / 1e9,
                    drive.free_bytes as f64 / 1e9,
                ));
            }
        });
    }
}
```

**关键点**:
- 在 `new()` 中调用 `enumerate_drives()`，结果存入 `self.drives`
- UI 中遍历展示每个盘的盘符和空间信息（GB 为单位，除以 1e9 近似）
- 不使用 `format!("{:.1}", bytes as f64 / 1024.0 / 1024.0 / 1024.0)` 也可以，1e9 足够直观
